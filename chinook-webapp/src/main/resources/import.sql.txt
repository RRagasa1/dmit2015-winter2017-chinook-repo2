CREATE TABLE APP_USER
(
	loginName	VARCHAR(64) PRIMARY KEY NOT NULL,
	password	VARCHAR(64) NOT NULL
);

CREATE TABLE APP_USER_ROLE
(
	loginName	VARCHAR(64) NOT NULL,
	roleName	VARCHAR(32) NOT NULL,
	PRIMARY KEY(loginName, roleName)
);

ALTER TABLE APP_USER_ROLE ADD CONSTRAINT FK_Loginname FOREIGN KEY(loginName) REFERENCES APP_USER(loginName);
